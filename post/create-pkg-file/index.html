Create .pkg file for macOS.

You can use quickpkg, https://github.com/scriptingosx/quickpkg

For ProjectA, create pkg file on macOS.

export VERSION=1.0
export RELEASE=1
export PROJECT_NAME=ProjectA
export CURRENT_DIR=$(cd $(dirname $0); pwd)
cd ../
export PROJECT_PATH=$(cd $(dirname $0); pwd)
export PKG_CMD=${PROJECT_PATH}/../quickpkg-0.8.1/quickpkg

export APP_BUNDLE_FOLDER=./bin/${PROJECT_NAME}.app
if [ ! -d "${APP_BUNDLE_FOLDER}" ]; then
    cp -r ./bin/Template.app ${APP_BUNDLE_FOLDER}
fi

#copy Info.plist, ${PROJECT_NAME}.icns 
cp ./doc/${PROJECT_NAME}/Info.plist ./bin/${PROJECT_NAME}.app/Contents/Info.plist
cp ./icon/${PROJECT_NAME}.icns ./bin/${PROJECT_NAME}.app/Contents/Resources/
cd bin
rm -f ./${PROJECT_NAME}.app/Contents/MacOS/*.*
cp ./${PROJECT_NAME} ./${PROJECT_NAME}.app/Contents/MacOS/
${PKG_CMD} ./${PROJECT_NAME}.app --output ../dist/macos/${PROJECT_NAME}-${VERSION}.${RELEASE}.pkg 
cd ${CURRENT_DIR}
